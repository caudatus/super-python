for i in range(10):
    if i % 2 == 0:
        print(i)

# 코드 B
for i in range(0, 10, 2):
    print(i)

코드 A: 0~9를 전부 순회하면서 짝수만 걸러냄 (10번 반복)
코드 B: 처음부터 2씩 건너뛰어 짝수만 방문 (5번 반복)

🏆 도전 문제 — 아래 리스트의 점수들로 평균을 계산하고, 평균 이상인 점수 옆에 ⭐를 붙여 출력하는 코드를 작성하세요.

scores = [72, 88, 65, 91, 79, 84]
출력 예시:72 / 88 ⭐ / 65 / 91 ⭐ / 79 / 84 ⭐
💡 힌트
# 1단계: 합계 구하기
total = 0
for s in scores:
    total += s
avg = total / len(scores)   ← 평균 계산

# 2단계: 평균 이상이면 ⭐ 붙여 출력
for s in scores:
    if s >= avg:
        print(f"{s} ⭐")
    else:
        print(s)
✅ 정답
scores = [72, 88, 65, 91, 79, 84]

# 1단계: 합계 → 평균 계산
total = 0
for s in scores:
    total += s
avg = total / len(scores)   # 79.83...

# 2단계: 평균 이상이면 ⭐
for s in scores:
    if s >= avg:
        print(f"{s} ⭐")
    else:
        print(s)
# 평균(79.83) 이상: 88, 91, 84
