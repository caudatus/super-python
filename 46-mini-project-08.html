STEP 1 파일에서 일기 불러오기 파일 읽기 예외처리 ►

load_diary(filename) 함수를 만드세요. diary.txt를 읽어 --- 구분선으로 각 일기를 분리하고, 각 블록의 첫 줄 DATE: ...에서 날짜를 추출해 딕셔너리 리스트로 반환합니다.

파일이 없으면 FileNotFoundError를 잡아 빈 리스트를 반환하세요. 이게 핵심입니다 — 처음 실행할 때 파일이 없어도 앱이 죽으면 안 됩니다.

def load_diary(filename):
    entries = []
    try:
        with open(filename, "r", encoding="utf-8") as f:
            raw = f.read()
        blocks = raw.strip().split("---")   # --- 로 분리
        for block in blocks:
            block = block.strip()
            if not block:
                continue
            lines = block.splitlines()
            date = lines[0].replace("DATE: ", "")   # 첫 줄에서 날짜 추출
            content = "\n".join(lines[1:]).strip()
            entries.append({"date": date, "content": content})
    except FileNotFoundError:
        pass   # 파일 없으면 빈 리스트 그대로 반환
    return entries

def save_entry(filename, date, content):
    try:
        with open(filename, "a", encoding="utf-8") as f:
            f.write(f"DATE: {date}\n")
            f.write(content + "\n")
            f.write("---\n")
        print(f"  ✅ 일기가 저장되었습니다 ({filename})")
    except IOError as e:
        print(f"  ❌ 저장 실패: {e}")

def view_all(entries):
    if not entries:
        print("\n  아직 작성된 일기가 없습니다")
        return
    print(f"\n[전체 일기 목록] — {len(entries)}건")
    for i, entry in enumerate(entries, 1):
        print("━" * 40)
        print(f"[{i}] {entry['date']}")
        print(entry['content'])
    print("━" * 40)

STEP 5 삭제 함수 + 메뉴 루프 조립 파일 재저장 ValueError ►

delete_entry(entries, filename) — view_all()로 목록을 보여주고, 번호를 입력받아 entries에서 해당 항목을 삭제합니다. 이후 남은 항목 전체를 파일에 다시 씁니다("w" 모드). 잘못된 번호 입력은 try/except로 처리하세요.

main() — while True 루프로 메뉴를 반복 출력하고, 선택에 따라 각 함수를 호출합니다. 5번 종료 선택 시 break로 루프를 탈출하세요.

def delete_entry(entries, filename):
    view_all(entries)
    if not entries:
        return
    try:
        num = int(input("삭제할 번호: "))
        if not (1 <= num <= len(entries)):
            raise ValueError("범위를 벗어났습니다")
        removed = entries.pop(num - 1)
        # 파일 전체 재저장
        with open(filename, "w", encoding="utf-8") as f:
            for e in entries:
                f.write(f"DATE: {e['date']}\n{e['content']}\n---\n")
        print(f"  ✅ [{removed['date']}] 삭제 완료")
    except ValueError as e:
        print(f"  ❌ {e}")

def main():
    FILENAME = "diary.txt"
    entries = load_diary(FILENAME)
    while True:
        # 메뉴 출력 후 선택 받기

        elif choice == "3": search_diary(entries)
        elif choice == "4": delete_entry(entries, FILENAME)
        elif choice == "5": break
        else: print("  1~5 중 선택해주세요")

 39    print("오늘의 일기를 쓰세요 (빈 줄 입력으로 저장):")
 40    lines = []
 41    while True:
 42        line = input()
 43        if line == "":
 44            break
 45        lines.append(line)
 46    content = "\n".join(lines)
 47    if not content.strip():
 48        print("  내용이 없어 저장하지 않았습니다")
 49        return
 50    save_entry(filename, date, content)
 51    entries.append({"date": date, "content": content})

 53def view_all(entries):
 54    if not entries:
 55        print("\n  아직 작성된 일기가 없습니다")
 56        return
 57    print(f"\n[전체 일기 목록] — {len(entries)}건")
 58    for i, entry in enumerate(entries, 1):
 59        print(SEP)
 60        print(f"[{i}] {entry['date']}")
 61        print(entry['content'])
 62    print(SEP)
 63
 64def search_diary(entries):
 65    keyword = input("\n검색할 날짜를 입력하세요 (예: 2026-03): ").strip()
 66    results = [e for e in entries if keyword in e["date"]]
 67    if not results:
 68        print(f"\n  '{keyword}' 검색 결과가 없습니다")
 69        return
 70    print(f"\n  검색 결과 {len(results)}건")
 71    for entry in results:
 72        print(SEP)
 73        print(f"[{entry['date']}]\n{entry['content']}")
 74    print(SEP)
 75
 76def delete_entry(entries, filename):
 77    view_all(entries)
 78    if not entries:
 79        return
 80    try:
 81        num = int(input("\n삭제할 번호를 입력하세요: "))
 82        if not (1 <= num <= len(entries)):
 83            raise ValueError(f"1~{len(entries)} 사이의 번호를 입력해주세요")
 84        removed = entries.pop(num - 1)
 85        with open(filename, "w", encoding="utf-8") as f:
 86            for e in entries:
 87                f.write(f"DATE: {e['date']}\n{e['content']}\n---\n")
 88        print(f"  ✅ [{removed['date']}] 삭제 완료")
 89    except ValueError as e:
 90        print(f"  ❌ {e}")
